## PROPOSED CATEGORY SCHEMA for `/moje-vencanje`

```typescript
type VendorCategory =
  | "sala" // 47 vendors
  | "bend" // 32 vendors (bands + orchestras + solo)
  | "dj" // 10 vendors
  | "fotograf" // 22 vendors
  | "snimatelj" // 8 vendors (some overlap with fotograf)
  | "torta" // 22 vendors
  | "dekoracija" // 18 vendors
  | "cvece" // 20 vendors (flowers + bouquets)
  | "vatromet" // 4 national vendors
  | "efekti" // 14 vendors (fog, confetti, sparklers)
  | "vencanica" // 25 vendors
  | "sminkanje" // 12 vendors (MUA + styling)
  | "burme" // 16 vendors
  | "pokloni"; // 10 vendors

type City =
  | "beograd"
  | "novi-sad"
  | "subotica"
  | "cacak"
  | "kragujevac"
  | "nis";

// Base fields for all vendors
interface Vendor {
  name: string; // NAME — display name
  slug: string; // URL-safe identifier (auto-generated from name)
  category: VendorCategory;
  city: City;
  phone: string | null; // PHONE
  website: string | null; // WEBSITE
  instagram: string | null; // IG — handle without @
  verified: boolean; // manual verification flag
}

// Category-specific extensions
interface VenueVendor extends Vendor {
  category: "sala";
  address: string | null; // ADDRESS — full street address
  capacity: string | null; // CAPACITY — e.g. "200-400", "300 / 120 terasa"
}

interface MusicVendor extends Vendor {
  category: "bend" | "dj";
  type: "bend" | "orkestar" | "solo" | "dj" | "duo";
  genres: string[]; // e.g. ["pop", "folk", "zabavna"]
  memberCount?: number;
}

interface PhotographyVendor extends Vendor {
  category: "fotograf" | "snimatelj";
  style: string[]; // e.g. ["dokumentarni", "umetnički"]
  offersVideo: boolean;
  offersPhoto: boolean;
}

interface CakeVendor extends Vendor {
  category: "torta";
  address: string | null;
  specialty: string[]; // e.g. ["fondant", "naked cake", "čokoladne"]
  pricePerKg?: string; // e.g. "1800 RSD/kg"
}

interface DecorationVendor extends Vendor {
  category: "dekoracija" | "cvece" | "vatromet" | "efekti";
  services: string[]; // e.g. ["stolne dekoracije", "bidermajer", "teški dim"]
  coversAllSerbia: boolean;
}

interface BridalVendor extends Vendor {
  category: "vencanica" | "sminkanje" | "burme" | "pokloni";
  address: string | null;
  brands?: string[]; // e.g. ["Pronovias", "MillaNova"]
  priceRange?: string; // e.g. "200-600 EUR"
}
```
